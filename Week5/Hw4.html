val increment : int list -> int list

increment list is the given list of ints with each element increased by 1.

val lengths : 'a list list -> int list

lengths list is a list of the lengths of each list in list.

val times_list : int -> int list -> int list

times_lists x list is a list of each element of list multiplied by num.

val product : int list -> int

product list is the product of all elements in list.

val print_it : string list -> unit

print_it list prints each element of list to a new line.

val intercalate : string -> string list -> string

intercalate sep list separates each element of list with sep. Ignores empty strings.

val nonempty : 'a list list -> 'a list list

nonempty list is all nonempty lists in list in the same order.

val odds : int list -> int list

odds list is a list of all odd numbers in list.

val skip_skips : string list -> string list

skip_skips list is all strings in list excluding any strings that equal "skip".

val flip : 'a tree -> 'a tree

flip tree is tree with each set of left and right nodes swapped.

val exists_tree : ('a -> bool) -> 'a tree -> bool

exists_tree f tree is true if a node in tree satisfies f and false if no node in tree satisfies f.

val fold_tree : 'a -> ('b -> 'a -> 'a -> 'a) -> 'b tree -> 'a

fold_tree init f tree is tree folded using f.

val map_tree : ('a -> 'b) -> 'a tree -> 'b tree

map_tree f tree is f applied to each node in tree.

val for_all_tree : ('a -> bool) -> 'a tree -> bool

for_all_tree is true if and only if f applied to each node in tree is true. Otherwise, is false.
